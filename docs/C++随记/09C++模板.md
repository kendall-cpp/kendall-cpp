
1.模板设置类型参数


```cpp
//模板
#include <iostream>
using std::cout;
using std::endl;

template <class T>
T add(T a,T b) {
    return a + b;
}

int main() {

    int a = 1,b = 2;
    double d1 = 1.1,d2 = 2.2;
    char ch1 = 'a',ch2 = 2;
    cout << "a + b = " << add(a,b) << endl;   //3
    cout << "d1 + d2 = " << add(d1,d2) << endl;  //3.3
    cout << "ch1 + ch2 = " << add(ch1,ch2) << endl;  //c
    return 0;
}
```

4.可变参数模板参数 模板参数包
```cpp
#include <iostream>
using std::cout;
using std::endl;

//可变参数
template<typename... Args>
void print(Args... args){
    //计算可变参数的个数
    cout << "sizeof...(Args) = " << sizeof...(Args) << endl;
    cout << "sizeof...(args) = " << sizeof...(args) << endl;
}

int main() {

    int a = 1;
    double b = 2.2;
    char ch = 'a';
    print(a,b,ch);
    return 0;
}
```
