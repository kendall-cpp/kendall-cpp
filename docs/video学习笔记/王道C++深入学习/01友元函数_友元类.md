## 1.友元函数和友元类

一般来说，类的私有成员只能在类的内部访问，类外的函数是不能访问它们的。但是，可以将一个函数定义为类的**友元函数**，这时该函数就可以访问该类的私有成员了。同时，也可以把另一个类B定义为本类A的友元类，这样B类就可以访问A类的任何成员了。

### 1.1 友元函数之普通函数

```cpp
#define _CRT_SECURE_NO_WARNINGS 1 
#include <iostream> 
#include <math.h>

using namespace std;

class Point
{
public:
	//构造函数
	Point(int x = 0, int y = 0)
	:_ix(x)
	,_iy(y)
	{
		cout << "Point(int x,int y)" << endl;
	}
	void print() const
	{
		cout << "(" << _ix
			<< "," << this->_iy
			<< ")";
	}
	//求两点之间的距离
	//将distance设置为友元  
	//将普通函数设置为友元函数（非成员函数）
	friend float distance(const Point& lhs, const Point & rhs);
	
private:
	int _ix;
	int _iy;
};

//计算两点之间的距离
//设置成友元函数才能使用Point里面的私有变量
float distance(const Point& lhs, const Point& rhs)
{
	return sqrt((lhs._ix - rhs._ix) * (lhs._ix - rhs._ix) +
		(lhs._iy - rhs._iy) * (lhs._iy - rhs._iy));
}

//测试友元函数
void test01()
{
	Point pt1(1, 2), pt2(3,4);
	pt1.print();
	cout << "到";
	pt2.print();
	cout << "的距离是: " << distance(pt1, pt2) << endl;
}
```

### 1.3 友元类

友元是单向的，并且不具备传递性，友元不能继承

```cpp
#define _CRT_SECURE_NO_WARNINGS 1 
#include <iostream> 
#include <math.h>

using namespace std;

//需要先声明Point
class Point;

class Line
{
public:
	float distance(const Point& lhs, const Point& rhs);
	void setPoint(Point *pt,int ix,int iy);

//在Line里面顶一个私有成员，在Point是不能更改的，因为友元是单向的
private:
	int _iz;
};

class Point
{
public:
	//构造函数
	Point(int x = 0, int y = 0)
	:_ix(x)
	,_iy(y)
	{
		cout << "Point(int x,int y)" << endl;
	}
	void print() const
	{
		cout << "(" << _ix
			<< "," << this->_iy
			<< ")";
	}
	
	//Point试图去修改line的成员（不能修改,会直接报错）
	// void setLine(Line *lin,int iz)
	// {
	// 	lin._iz = iz;
	// }

	//设置Line为友元类
	friend Line;
	
private:
	int _ix;
	int _iy;
};

//计算两点之间的距离
float Line::distance(const Point& lhs, const Point& rhs)
{
	return sqrt((lhs._ix - rhs._ix) * (lhs._ix - rhs._ix) +
		(lhs._iy - rhs._iy) * (lhs._iy - rhs._iy));
}
//通过Line去修改Point的成员
void Line::setPoint(Point * pt,int ix,int iy)
{
	pt->_ix = ix;
	pt->_iy = iy;
}

//测试友元函数
void test01()
{
	Point pt1(1, 2), pt2(3,4);
	pt1.print();
	cout << "到";
	pt2.print();
	cout << "的距离是: " << Line().distance(pt1, pt2) << endl;

	// 使用Line修改Point
	Line().setPoint(&pt1,5,6);
	cout << "修改后的pt1是：" ;
	pt1.print();
	cout << endl;
}

int main()
{
	test01();
	return 0;
}

```
执行结果：

```
Point(int x,int y)
Point(int x,int y)
(1,2)到(3,4)的距离是: 2.82843
修改后的pt1是：(5,6)
```

## 2.运算符重载

> C++ 特有的

```
不能重载的运算符有：
	成员访问符 .
	成员指针访问运算符 .*
	域运算符 ::
	长度运算符 sizeof	注意 sizeof 是一个运算符
	条件运算符号 ?:
```

**注意**: 
- sizeof 是一个运算符

```cpp
int main()
{
	int a = 10;
	printf("sizeof(a) = %d\n",sizeof a); //括号是可以拿掉的，类似 int a,只是这样写会出现警告，编译器不建议这样写
	printf("sizeof(a) = %d\n",sizeof(a));
	return 0;
}
```

- 运算符重载时，其参数必须是自定义类类型或者是枚举类型

### 2.1 普通函数进行重载

```cpp
//复数
#include <iostream>
using namespace std;

//定义一个复数
class Complex
{
public:
	Complex(double real = 0,double imag = 0)
	:_real(real)
	,_imag(imag)
	{
		cout << "this is Complex(double real = 0,double imag = 0)" << endl;
	}
	
	//打印复数
	void display()
	{
		// raal + 2imag 或者  a - 3imag
		cout << "这个复数是：";
		cout << _real;
		if(_imag  > 0)
			cout << " + " << _imag << "i" << endl;
		else if(_imag < 0)
			cout << " - " << (-1) * _imag << "i" << endl;
		else
			cout << endl;
	}

	//inline函数
	double getReal() const
	{
		return _real;
	}
	double getImag() const
	{
		return _imag;
	}


private:
	//实部和虚部
	double _real;
	double _imag;
};

//以普通函数进行重载 +
Complex operator+(const Complex & lhs,const Complex & rhs)
{
	return Complex(lhs.getReal() + rhs.getReal(),lhs.getImag() + rhs.getImag());

}

int main() 
{
	Complex c1(1,-2);
	Complex c2(3,4);
	c1.display();

	//假如现在要实现一个
	Complex c3 = c1 + c2;   //+ 没有被重载是不能用在自定义类型相加的，所以需要先重载运算符+
	c3.display();
	return 0;
}
```

### 2.2 以友元函数的形式重载

> 就不需要定义inline函数了

```cpp
class Complex
{
public:
	Complex(double real = 0,double imag = 0)
	:_real(real)
	,_imag(imag)
	{
		cout << "this is Complex(double real = 0,double imag = 0)" << endl;
	}
	
	//打印复数
	void display()
	{
		// raal + 2imag 或者  a - 3imag
		cout << "这个复数是：";
		cout << _real;
		if(_imag  > 0)
			cout << " + " << _imag << "i" << endl;
		else if(_imag < 0)
			cout << " - " << (-1) * _imag << "i" << endl;
		else
			cout << endl;
	}

	friend Complex operator+(const Complex & lhs,const Complex & rhs);

private:
	//实部和虚部
	double _real;
	double _imag;
};

//以友元函数进行重载 +
Complex operator+(const Complex & lhs,const Complex & rhs)
{
	return Complex(lhs._real + rhs._real,lhs._imag + rhs._imag);

}
```
